```python
import numpy as np
from typing import Dict

def get_gross_returns(net_returns: Dict[str, float]) -> np.ndarray:
    # Convert net returns from percentage (denoted by decimals) to gross returns
    # Gross return = 1 + Net return (in decimals)
    gross_returns = np.array([1 + (value / 100.0) for value in net_returns.values()])
    return gross_returns

def get_geometric_mean(gross_returns: np.ndarray) -> float:
    # Calculate the geometric mean of the gross returns
    return np.prod(gross_returns) ** (1 / len(gross_returns))

def get_net_average(gross_average: float) -> float:
    # Convert gross average back to net average
    # Net average = (Gross average - 1) * 100 (to convert to percentage)
    return (gross_average - 1) * 100

def get_average_return(
    net_returns: Dict[str, float],
) -> float:
    gross_returns: np.ndarray = get_gross_returns(net_returns)
    gross_average: float = get_geometric_mean(gross_returns)
    net_average: float = get_net_average(gross_average)
    return net_average
```

2. **get_geometric_mean**: This function calculates the geometric mean of an array of gross returns by taking the product of all elements raised to the power of \( \frac{1}{n} \), where \( n \) is the number of returns.

3. **get_net_average**: This function converts the calculated gross average back to the net percentage by applying the inverse of the gross return transformation.
